الحل A — Side Channel (Telegram-as-Output)

أعمل bash function صغيرة يرسل كل ردّي على Telegram قبل ما يطبع في الـ console.

# ضع في ~/.bashrc أو قبل كل أمر
export TELEGRAM_BOT_TOKEN="8483274654:AAE1zwTH4Jst8GX9vvpETyjFzlZVN-DijE8"
export TELEGRAM_CHAT_ID="2095951462"

tg-reply() {
  local text="$1"
  curl -s -X POST "https://api.telegram.org/bot$TELEGRAM_BOT_TOKEN/sendMessage" \
    -d "chat_id=$TELEGRAM_CHAT_ID" \
    -d "text=$text" \
    -d "parse_mode=HTML" > /dev/null
}

# الاستخدام:
tg-reply "🎉 الرد جاهز يا Anas!"

المميزات

  • فوري، بدون setup معقد
  • ما يحتاج server عام

العيوب

  • يتطلب تدخل يدوي

الحل B — Webhook (الأقوى)

خادم صغير يستقبل Telegram webhooks ويربطها بـ Mavis session.

# server.py (Flask)
from flask import Flask, request
import requests, os

app = Flask(__name__)
TOKEN = os.getenv('TELEGRAM_BOT_TOKEN')
CHAT_ID = os.getenv('TELEGRAM_CHAT_ID')

@app.route('/webhook', methods=['POST'])
def webhook():
    data = request.get_json()
    msg = data.get('message', {})
    chat_id = msg.get('chat', {}).get('id')
    text = msg.get('text', '')
    
    # Forward to Mavis session
    mavis_reply = forward_to_mavis(text)
    
    # Send reply to Telegram
    requests.post(f"https://api.telegram.org/bot{TOKEN}/sendMessage",
        json={"chat_id": chat_id, "text": mavis_reply})
    return '', 200

# Run + set webhook:
# 1. Deploy server to VPS / Cloudflare Worker
# 2. curl -X POST "https://api.telegram.org/bot$TOKEN/setWebhook?url=https://yourdomain.com/webhook"

المميزات

  • أتمتة كاملة (incoming + outgoing)
  • Real-time
  • production-ready
  • يدعم multiple users

العيوب

  • محتاج VPS / Cloudflare Worker
  • محتاج domain + SSL
  • setup أعقد (10-15 دقيقة)
  • محتاج uptime monitoring

الحل C — Poll Loop

Script يدور في الخلفية يسأل Telegram عن رسائل جديدة.

# poll.sh - run in background
LAST_ID=0

while true; do
  updates=$(curl -s "https://api.telegram.org/bot$TOKEN/getUpdates?offset=$((LAST_ID+1))&timeout=30")
  echo "$updates" | python3 -c "
import sys, json
data = json.load(sys.stdin)
for u in data.get('result', []):
    msg = u.get('message', {})
    print(f\"NEW: {msg.get('text')} from {msg.get('from', {}).get('first_name')}\")
    global LAST_ID
    LAST_ID = max(LAST_ID, u['update_id'])
"
  sleep 2
done

المميزات

  • ما يحتاج webhook URL
  • بسيط للنشر

العيوب

  • مشكلة 409 Conflict لو في instance آخر
  • ممكن يفقد رسائل لو تعطّل
  • ما ينفع مع الـ setup الحالي (instance conflict)
